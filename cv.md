### Example of code: ###

*Task from manual LearnJavaScript, subject: arrays:*

*Write a function that asks the user to enter values using prompt and stores them in an array.
It stops asking for values when the user enters a non-numeric value, an empty string, or presses "Cancel".
It calculates and returns the sum of the array elements*
```
function sumInput() {
let numbers = [];
while (true) {
  let value = prompt("Enter a number");
  // stop input
  if (value === "" || value === null) {  // do not use a gap between "" 
    break;
  } 
  numbers.push (+value); // add elements in the array
}
let sum = 0; // calculate the sum of all elements
for( let number of numbers){
    sum += number;
}
return sum;
}
alert( sumInput() ); // alert the total amount
```
